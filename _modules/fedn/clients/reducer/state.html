    """

    :param state:
    :return:
    """
    if state == ReducerState.setup:
        return "setup"
    if state == ReducerState.idle:
        return "idle"
    if state == ReducerState.instructing:
        return "instructing"
    if state == ReducerState.monitoring:
        return "monitoring"

    return "UNKNOWN"
